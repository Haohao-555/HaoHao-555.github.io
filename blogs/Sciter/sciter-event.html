    # # 向 窗口层（Window）发布事件

    # 1、父与子之间事件传递

    <div class="container"></div>
<script type="module">
class App extends Element {
    constructor() {
        super()
    }

    componentDidMount() {
        // 订阅 窗口事件名为 do-modal
        document.onGlobalEvent("do-modal", evt => {
           console.log('------------------- App -------------------');
           console.log('事件类型', evt.type);
           console.log('事件传递的参数', evt.data);
           console.log('-------------------------------------------');
        });
    }

    render() {
        return (
            <div class="container">
                <ChildOne />
            </div>
        )
    }
}
document.$('.container').content(<App />);
</script>

    # # 总结

    1、向 窗口层（Window）发布事件

    • 通过：Window.post(new Event("自定义事件名称", { data: '你的数据' })); 发布事件
    • 通过：document.onGlobalEvent("自定义事件名称", evt => { }); 订阅事件
    • 事件的订阅与发布是面向所有窗口的，只要其他窗口订阅了该自定义事件就会触发

    2、 向 Document 层 发布事件

    • 通过：document.postEvent(new Event("自定义事件名称", { data: '你的数据' } })); 发布事件
    • 通过：document.on('自定义事件名', evt => { }); 订阅事件
    • 通过：document.off('自定义事件名'); 取消事件订阅
    • 事件的订阅与发布是面向当前窗口的，其他窗口订阅了该自定义事件也不会触发

<script type="module">
class ChildOne extends Element {
    constructor() {
        super();
    }

    render() {
        return (
            <button class="do">ChildOne</button>
        )
    }

    // 给 类名为 do 的 button 添加 click 事件 
    ["on click at button.do"](evt, button) {
        // console.log(this, document); // this 是 事件触发的对象

        // 方式一：向 所有 Window 层 抛出事件名称为 do-modal 
        // Window.post(new Event("do-modal", { data: { content: 'ChildOne' } }));
        // Window.post(new Event("do-modal-ChildTwo", { data: { content: 'ChildOne to ChildTwo' } }));


        // 方式二：向 document 层 中 app 组件 抛出事件名称 为 do-modal.App
        // document.postEvent(new Event("do-modal.App", { data: { content: 'ChildOne to App' } }));

        // 向兄弟组件（ChildTwo）传递 事件名称为 do-modal.ChildTwo
        document.postEvent(new Event("do-modal.ChildTwo", { data: { content: 'ChildOne to ChildTwo' } }));
    }
}

class ChildTwo extends Element {
    constructor() {
        super();
    }

    componentDidMount() {

        // document 订阅事件 do-modal.ChildTwo
        document.on('do-modal.ChildTwo', evt => {
            console.log('------------------- ChildTwo -------------------');
            console.log('监听到事件类型', evt.type);
            console.log('事件传递的参数', evt.data);
            console.log('------------------------------------------------');
        });

        // // 定阅事件 do-modal.ChildTwo 全局
        // document.onGlobalEvent("do-modal-ChildTwo", evt => {
        //     console.log('------------------- ChildTwo -------------------');
        //     console.log('事件类型', evt.type);
        //     console.log('事件传递的参数', evt.data);
        //     console.log('------------------------------------------------');
        // });
    }

    render() {
        return (
            <button class="do">ChildTwo</button>
        )
    }

    // 给 类名为 do 的 button 添加 click 事件 
    ["on click at button.do"](evt, button) {
        // console.log(this, document); // this 是 事件触发的对象

        // 方式一：向 Window 层 抛出事件名称为 do-modal 
        Window.post(new Event("do-modal", { data: { content: 'ChildTwo' } }));



        // 方式二： 向 document 层 中 app 组件 抛出事件名称 为 do-modal
        // document.postEvent(new Event("do-modal.App", { data: { content: 'ChildTwo to App' } }));
    }
}

class App extends Element {
    constructor() {
        super()
    }

    componentDidMount() {
        // 对应 子组件 方式一 （订阅事件）
        // document.onGlobalEvent("do-modal", evt => {
        //     console.log('------------------- App -------------------');
        //     console.log('事件类型', evt.type);
        //     console.log('事件传递的参数', evt.data);
        //     console.log('-------------------------------------------');
        // });

        // 对应 子组件 方式二 （订阅事件）
        document.on('do-modal.App', evt => {
            console.log('------------------- App -------------------');
            console.log('监听到事件类型', evt.type);
            console.log('事件传递的参数', evt.data);
            console.log('-------------------------------------------');
        });
    }

    render() {
        return (
            <div class="container">
                <ChildOne />
                <ChildTwo />
            </div>
        )
    }
}
document.$('.container').content(<App />);
</script>
